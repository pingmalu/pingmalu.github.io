# scp拷贝最近3天日期命名的tar.z文件，并删除本地6天前的tar.z文件

	#!/bin/bash
	for((i=0;i<=3;i++)); do
		datefile=`date -d"$i day ago" +%Y%m%d`.tar.z
		if [ ! -f "$datefile" ]; then
			scp malu.me:/backup/$datefile /home/backup/
		fi
	done
	find ./ -mtime +6 -type f -name "*.tar.z"|xargs rm -f
